---
title: "Създаване на lead"
api: "POST /user/lead"
description: "Създаване на нов lead в системата на Aiplocalls"
---

Този endpoint ви позволява да създадете нов lead в системата на Aiplocalls.

### Request body

<ParamField body="phone_number" type="string" required>
  Телефонният номер на lead-а в E.164 формат (например +1234567890)
</ParamField>

<ParamField body="campaign_id" type="integer" required>
  ID-то на кампанията, за която да се създаде lead-ът
</ParamField>

<ParamField body="variables" type="object">
  Персонализирани променливи за предаване към lead-а. Можете да включите **всички променливи, които са дефинирани във вашия асистент** (вижте [Променливи за повиквания](/ai-assistants/settings/prompt-and-tools#call-variables)). Примерите по-долу са само за илюстрация — използвайте вашите собствени имена на променливи, както са конфигурирани в настройките на асистента.

  <Expandable title="Примерни променливи (добавете вашите собствени)">
    <ParamField body="customer_name" type="string">
      например името на клиента
    </ParamField>

    <ParamField body="email" type="string">
      например имейлът на клиента
    </ParamField>
  </Expandable>
</ParamField>

<ParamField body="allow_dupplicate" type="boolean">
  Дали да се позволяват дублиращи се leads в кампанията.
</ParamField>

<ParamField body="secondary_contacts" type="array">
  Масив от secondary contact leads, които да се създадат заедно с основния lead
  
  <Expandable title="свойства на secondary_contacts">
    <ParamField body="phone_number" type="string" required>
      Телефонният номер на secondary contact-а в E.164 формат
    </ParamField>
    
    <ParamField body="variables" type="object">
      Променливите за предаване към secondary contact-а
      
      <Expandable title="свойства на variables">
        <ParamField body="customer_name" type="string" default="<string>">
          Името на secondary contact-а
        </ParamField>
        
        <ParamField body="email" type="string" default="<string>">
          Имейлът на secondary contact-а
        </ParamField>
      </Expandable>
    </ParamField>
  </Expandable>
</ParamField>

### Response

<ResponseField name="message" type="string" default="Lead created successfully">
  Съобщението от отговора
</ResponseField>

<ResponseField name="data" type="object">
  Обектът с данни за lead-а
  
  <Expandable title="свойства на data">
    <ResponseField name="id" type="integer">
      Уникалното ID на създадения lead
    </ResponseField>
    
    <ResponseField name="campaign_id" type="integer">
      ID-то на кампанията, към която принадлежи този lead
    </ResponseField>

    <ResponseField name="phone_number" type="string">
      Телефонният номер на lead-а в E.164 формат
    </ResponseField>
    
    <ResponseField name="variables" type="object">
      Променливите, свързани с lead-а
      
      <Expandable title="свойства на variables">
        <ResponseField name="customer_name" type="string" default="<string>">
          Името на клиента
        </ResponseField>
        
        <ResponseField name="email" type="string" default="<string>">
          Имейлът на клиента
        </ResponseField>
      </Expandable>
    </ResponseField>
    
    <ResponseField name="status" type="string" default="created">
      Статусът на lead-а
    </ResponseField>
    
    <ResponseField name="created_at" type="string">
      Времевият отпечатък, когато е създаден lead-ът
    </ResponseField>
    
    <ResponseField name="updated_at" type="string">
      Времевият отпечатък, когато lead-ът е актуализиран за последно
    </ResponseField>

    <ResponseField name="campaign" type="object">
      Информацията за кампанията
      <Expandable title="свойства на campaign">
        <ResponseField name="id" type="integer">
          ID-то на кампанията
        </ResponseField>
        <ResponseField name="name" type="string">
          Името на кампанията
        </ResponseField>
      </Expandable>
    </ResponseField>

    <ResponseField name="secondary_contacts" type="array">
      Масив от secondary contact leads, свързани с този lead
      <Expandable title="свойства на secondary_contacts">
        <ResponseField name="id" type="integer">
          ID-то на secondary contact lead-а
        </ResponseField>
        <ResponseField name="phone_number" type="string">
          Телефонният номер на secondary contact-а в E.164 формат
        </ResponseField>
        <ResponseField name="variables" type="object">
          Променливите, свързани с secondary contact-а
        </ResponseField>
        <ResponseField name="status" type="string">
          Статусът на secondary contact-а
        </ResponseField>
        <ResponseField name="created_at" type="string">
          Датата на създаване на secondary contact-а
        </ResponseField>
        <ResponseField name="updated_at" type="string">
          Датата на актуализация на secondary contact-а
        </ResponseField>
      </Expandable>
    </ResponseField>
  </Expandable>
</ResponseField>

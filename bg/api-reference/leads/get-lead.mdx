---
---
title: "Списък с потенциални клиенти"
api: "GET /user/leads"
description: "Извеждане на всички потенциални клиенти за удостоверения потребител с опции за филтриране и пагинация"
---

Този endpoint ви позволява да изведете всички потенциални клиенти, принадлежащи на удостоверения потребител, с различни опции за филтриране и пагинация.

### Query параметри

<ParamField query="status" type="string" optional>
  Филтриране на потенциални клиенти по статус. Възможни стойности: `created`, `scheduled`, `processing`, `completed`, `rescheduled`, `reached-max-retries`, `blacklisted`
</ParamField>

<ParamField query="campaign_id" type="integer" optional>
  Филтриране на потенциални клиенти по ID на кампания
</ParamField>

<ParamField query="phone_number" type="string" optional>
  Филтриране на потенциални клиенти по телефонен номер (поддържа се частично съвпадение)
</ParamField>

<ParamField query="date_from" type="string" optional>
  Филтриране на потенциални клиенти, създадени от тази дата (формат YYYY-MM-DD)
</ParamField>

<ParamField query="date_to" type="string" optional>
  Филтриране на потенциални клиенти, създадени до тази дата (формат YYYY-MM-DD)
</ParamField>

<ParamField query="per_page" type="integer" optional>
  Брой потенциални клиенти на страница (1-100, по подразбиране: 15)
</ParamField>

<ParamField query="page" type="integer" optional>
  Номер на страница (по подразбиране: 1)
</ParamField>

### Полета в отговора

<ResponseField name="data" type="array">
  <Expandable title="свойства">
    <ResponseField name="id" type="integer">
      ID на потенциалния клиент
    </ResponseField>

    <ResponseField name="campaign_id" type="integer">
      ID на кампанията, към която принадлежи този потенциален клиент
    </ResponseField>

    <ResponseField name="phone_number" type="string">
      Телефонният номер на потенциалния клиент във формат E.164
    </ResponseField>

    <ResponseField name="variables" type="object">
      Променливите, свързани с потенциалния клиент
    </ResponseField>

    <ResponseField name="status" type="string">
      Статусът на потенциалния клиент
    </ResponseField>

    <ResponseField name="created_at" type="string">
      Датата на създаване на потенциалния клиент
    </ResponseField>

    <ResponseField name="updated_at" type="string">
      Датата на актуализиране на потенциалния клиент
    </ResponseField>

    <ResponseField name="campaign" type="object">
      Информацията за кампанията
      <Expandable title="свойства на campaign">
        <ResponseField name="id" type="integer">
          ID на кампанията
        </ResponseField>
        <ResponseField name="name" type="string">
          Името на кампанията
        </ResponseField>
      </Expandable>
    </ResponseField>

    <ResponseField name="secondary_contacts" type="array">
      Масив от вторични контакти, свързани с този потенциален клиент
      <Expandable title="свойства на secondary_contacts">
        <ResponseField name="id" type="integer">
          ID на вторичния контакт
        </ResponseField>
        <ResponseField name="phone_number" type="string">
          Телефонният номер на вторичния контакт във формат E.164
        </ResponseField>
        <ResponseField name="variables" type="object">
          Променливите, свързани с вторичния контакт
        </ResponseField>
        <ResponseField name="status" type="string">
          Статусът на вторичния контакт
        </ResponseField>
        <ResponseField name="created_at" type="string">
          Датата на създаване на вторичния контакт
        </ResponseField>
        <ResponseField name="updated_at" type="string">
          Датата на актуализиране на вторичния контакт
        </ResponseField>
      </Expandable>
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="current_page" type="integer">
  Номерът на текущата страница
</ResponseField>

<ResponseField name="per_page" type="integer">
  Брой елементи на страница
</ResponseField>

<ResponseField name="total" type="integer">
  Общ брой потенциални клиенти, отговарящи на критериите
</ResponseField>

<ResponseField name="last_page" type="integer">
  Номерът на последната страница
</ResponseField>

---
title: "Списък на обажданията"
api: "GET /user/calls"
description: "Списък на всички обаждания за автентикирания потребител с опции за филтриране"
---

Този endpoint ви позволява да получите списък на всички обаждания принадлежащи на автентикирания потребител с различни опции за филтриране.

### Query Parameters

<ParamField query="status" type="string" optional>
  Филтриране на обажданията по статус. Възможни стойности: `initiated`, `ringing`, `busy`, `in-progress`, `ended`, `completed`, `ended_by_customer`, `ended_by_assistant`, `no-answer`, `failed`
</ParamField>

<ParamField query="type" type="string" optional>
  Филтриране на обажданията по тип. Възможни стойности: `inbound`, `outbound`, `web`
</ParamField>

<ParamField query="phone_number" type="string" optional>
  Филтриране на обажданията по телефонен номер на клиента
</ParamField>

<ParamField query="assistant_id" type="integer" optional>
  Филтриране на обажданията по ID на асистента
</ParamField>

<ParamField query="campaign_id" type="integer" optional>
  Филтриране на обажданията по ID на кампанията
</ParamField>

<ParamField query="date_from" type="string" optional>
  Филтриране на обаждания от тази дата (формат YYYY-MM-DD)
</ParamField>

<ParamField query="date_to" type="string" optional>
  Филтриране на обаждания до тази дата (формат YYYY-MM-DD)
</ParamField>

<ParamField query="per_page" type="integer" optional>
  Брой обаждания на страница (1-100, по подразбиране: 15)
</ParamField>

<ParamField query="page" type="integer" optional>
  Номер на страницата (по подразбиране: 1)
</ParamField>

### Response fields

<ResponseField name="data" type="array">
  <Expandable title="properties">
    <ResponseField name="id" type="integer">
      Уникалният идентификатор на обаждането
    </ResponseField>

    <ResponseField name="assistant_name" type="string">
      Името на асистента, който обработи обаждането
    </ResponseField>

    <ResponseField name="campaign_name" type="string">
      Името на кампанията, към която принадлежи това обаждане (ако е приложимо)
    </ResponseField>

    <ResponseField name="type" type="string">
      Типът на обаждането (`inbound`, `outbound`, или `web`)
    </ResponseField>

    <ResponseField name="duration" type="integer">
      Продължителността на обаждането в секунди
    </ResponseField>

    <ResponseField name="assistant_phone_number" type="string">
      Телефонният номер използван от асистента
    </ResponseField>

    <ResponseField name="client_phone_number" type="string">
      Телефонният номер на клиента
    </ResponseField>

    <ResponseField name="status" type="string">
      Текущият статус на обаждането
    </ResponseField>

    <ResponseField name="transcript" type="string">
      Транскриптът на разговора от обаждането
    </ResponseField>

    <ResponseField name="variables" type="object">
      Променливи събрани по време на обаждането
    </ResponseField>

    <ResponseField name="evaluation" type="object">
      Данни за оценка на представянето на обаждането
    </ResponseField>

    <ResponseField name="webhook_response" type="object">
      Отговор от конфигурирани webhooks
    </ResponseField>

    <ResponseField name="carrier_cost" type="number">
      Цената наложена от оператора за това обаждане
    </ResponseField>

    <ResponseField name="total_cost" type="number">
      Общата цена на обаждането включително всички такси
    </ResponseField>

    <ResponseField name="answered_by" type="string">
      Кой отговори на обаждането (`human`, `machine`, или `unknown`)
    </ResponseField>

    <ResponseField name="recording_url" type="string">
      URL към записа на обаждането (ако е налично и активирано)
    </ResponseField>

    <ResponseField name="created_at" type="string">
      Датата и времето когато обаждането е създадено
    </ResponseField>

    <ResponseField name="updated_at" type="string">
      Датата и времето когато обаждането е актуализирано последно
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="current_page" type="integer">
  Номерът на текущата страница
</ResponseField>

<ResponseField name="per_page" type="integer">
  Брой елементи на страница
</ResponseField>

<ResponseField name="total" type="integer">
  Общ брой обаждания отговарящи на критериите
</ResponseField>

<ResponseField name="last_page" type="integer">
  Номерът на последната страница
</ResponseField>
